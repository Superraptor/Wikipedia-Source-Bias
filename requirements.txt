# Toolforge build-service dependency set.
#
# The buildpack installs exactly this file. Keep it lean: a container gets 10GB
# of storage INCLUDING the image, and there are no GPUs on Toolforge, so the
# transformers/torch sentiment backend is deliberately excluded here. It lives
# in requirements-dev.txt; analysis.py falls back to the NLTK/lexicon path when
# transformers is absent.

# The analysis package itself, installed from ./pyproject.toml. This is what
# makes `from wikipedia_sources_bias import analyze_page` resolve regardless of
# the process working directory -- see the note in backend/app.py.
.

# Web tier
Flask==3.0.3
gunicorn==23.0.0

# ToolsDB.
# 1.2.0 or newer is REQUIRED, not just preferred. Older releases run
# `DEFAULT_USER = getpass.getuser()` at import time and guard it with
# `except (ImportError, KeyError)`. Toolforge containers run as a bare uid with
# no /etc/passwd entry and no USER/LOGNAME set, and Python 3.13 changed
# getpass.getuser() to raise OSError instead of KeyError -- so on 1.1.x the
# statement `import pymysql` itself raises, before any connection is attempted.
PyMySQL==1.2.0

# Pulled in transitively by the analysis package, pinned here for reproducible
# container builds.
requests==2.32.3
beautifulsoup4==4.12.3
nltk==3.9.1
